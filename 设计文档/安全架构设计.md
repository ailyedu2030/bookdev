**网络隔离策略**：

| 区域 | 访问来源 | 允许端口 | 加密要求 |
|------|---------|---------|---------|
| DMZ | 外部网络 | 443 (HTTPS) | TLS 1.3 |
| 内部网络 | DMZ (仅API Gateway) | 8080 | mTLS |
| 安全区域 | 内部网络 | 8200 (Vault) | mTLS + AppRole |

### 2.2 API Gateway防护

```yaml
api_gateway:
  waf_rules:
    - name: "SQL注入防护"
      condition: "sql_keywords_in_request"

    - name: "XSS防护"
      condition: "script_tags_or_events"
      action: "block"
```

## 三、L2 应用安全

### 3.1 输入验证（所有用户输入）

```yaml
input_validation:
  schemas:
    user_input:
      type: "string"
      min_length: 1
      max_length: 50000
      pattern: "^(?!.*[\\x00-\\x1f]).*$"  # 无控制字符
    
    material_content:
      type: "string"
      max_length: 10485760  # 10MB
      allowed_extensions: [".pdf", ".docx", ".txt", ".md", ".csv"]
      virus_scan: true
    
    chapter_content:
      type: "string"
      max_length: 2097152  # 2MB (约100万字)
      content_filter: "strict"
    
    api_query:
      type: "object"
      max_depth: 5
      max_keys: 50

  sanitization:
    html:
      enabled: true
      allowed_tags: ["p", "br", "b", "i", "em", "strong"]
      allowed_attrs: []
    
    markdown:
      enabled: true
      unsafe_links: "strip"
      unsafe_images: "block"
    
    sql:
      prevention: "parameterized_only"
    
    system_prompts:
     隔离区: true
      禁止动态执行: true
```

### 3.2 输出过滤（防止XSS/注入）

```yaml
output_filtering:
  content_security:
    csp_header: |
      default-src 'self';
      script-src 'self';
      style-src 'self' 'unsafe-inline';
      img-src 'self' data: https:;
      font-src 'self';
      connect-src 'self' https://api.openai.com;
      frame-ancestors 'none';
    
    x_frame_options: "DENY"
    x_content_type_options: "nosniff"
    x_xss_protection: "1; mode=block"
    referrer_policy: "strict-origin-when-cross-origin"
  
  xss_prevention:
    output_encoding:
      html_context: "htmlencode"
      javascript_context: "jsencode"
      url_context: "urlencode"
      css_context: "cssencode"
    
    dom_purify:
      enabled: true
      config:
        ALLOWED_TAGS: ["p", "br", "b", "i", "em", "strong", "a", "ul", "ol", "li"]
        ALLOWED_ATTR: ["href"]
        FORBID_TAGS: ["script", "style", "iframe", "object", "embed"]
        FORBID_ATTR: ["onerror", "onclick", "onload"]
  
  injection_prevention:
    ai_output_review:
      enabled: true
      block_patterns:
        - "重复内容检测"
        - "乱码/无效Unicode检测"
        - "越界Token模式检测"
```

## 五、L4 AI安全

### 5.1 Prompt注入防御

```yaml
prompt_injection_defense:
  input_validation:
    length_limit: 50000
    format_check: "utf8_only"
    
    sensitive_word_filter:
      enabled: true
      action: "flag_and_reject"
      custom_list: |
        - "忽略之前的指令"
        - "ignore previous instructions"
        - "你现在是"
        - "你现在扮演"
        - "system prompt"
        - "/reset"
        - "新身份"
    
    injection_pattern_detection:
      enabled: true
      patterns:
        - name: "指令覆盖"
          regex: "(忽略|忘记|去掉).*(指令|规则|限制)"
        - name: "角色扮演逃逸"
          regex: "(你是|我扮演).*(AI|助手|系统)"
        - name: "分隔符注入"
          regex: "---+\\s*(system|user|assistant)"
        - name: "编码逃逸"
          regex: "(eval|exec|compile)\\s*\\("
      
      action: "sanitize_or_reject"
  
  prompt_structure:
    system_prompt_isolation:
      enabled: true
      boundary_markers:
        system_delimiter: "████████████████████████████"
        user_content_marker: "=== 用户输入 ==="
      
      rules:
        - "user_input永远不包含system指令"
        - "分隔符不参与模型推理"
        - "禁止动态代码执行"
    
    instruction_signing:
      enabled: true
      chain:
        - "系统管理员签署核心指令"
        - "用户意图经过输入验证"
        - "AI服务验证指令签名"
      
      signature_algorithm: "HMAC-SHA256"
      key_source: "Vault"
  
  runtime_protection:
    output_monitor:
      enabled: true
      checks:
        - "是否尝试泄露system prompt"
        - "是否包含指令覆盖内容"
        - "是否包含恶意代码"
      
      action: "block_and_alert"
```

### 5.2 指令签名链

```yaml
instruction_signing_chain:
  architecture:
    ┌─────────────────────────────────────────────────────────────┐
    │                    指令签名链流程                            │
    ├─────────────────────────────────────────────────────────────┤
    │                                                             │
    │  1. System Prompt  ──→  HMAC签名  ──→  签名值A              │
    │                                                             │
    │  2. 用户输入    ──→  输入验证  ──→  哈希摘要B               │
    │                                                             │
    │  3. 签名值A + 哈希摘要B  ──→  组合签名  ──→  最终签名C       │
    │                                                             │
    │  4. 最终签名C  ──→  AI服务验证  ──→  有效/无效              │
    │                                                             │
    └─────────────────────────────────────────────────────────────┘
  
  implementation:
    hmac_algorithm: "SHA-256"
    key_derivation: "HKDF"
    signature_encoding: "base64"
    timestamp_tolerance: 300  # 5分钟内有效
  
  verification:
    at_ai_service: true
    failure_action: "reject_and_log"
    alert_on_failure: true
```
