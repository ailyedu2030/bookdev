### 1.2 敏感词检测技术对比

| 方案 | 准确率 | 速度 | 维护成本 | 适用场景 | 评分 |
|------|--------|------|----------|----------|------|
| **正则表达式** | 60-70% | <1ms | 高(需手动更新词库) | 固定格式检测 | 4/10 |
| **传统ML模型** | 75-85% | 10-50ms | 中 | 通用敏感词 | 6/10 |
| **大模型判断** | 85-95% | 500-2000ms | 低 | 复杂语境理解 | 8/10 |
| **混合方案** | 90-95% | 50-200ms | 中 | 生产环境推荐 | 9/10 |

**技术实现建议：**

```python
# 推荐架构：三级过滤
class ContentFilter:
    def __init__(self):
        self.regex_filter = RegexFilter()      # 第一级：快速过滤
        self.ml_filter = MLFilter()          # 第二级：语义检测
        self.llm_judge = LLMJudge()          # 第三级：复杂判断

    async def filter(self, text: str) -> FilterResult:
        # 1. 正则快速匹配已知敏感词
        regex_result = self.regex_filter.check(text)
        if regex_result.is_blocked:
            return FilterResult(blocked=True, reason=regex_result.match)

        # 2. ML模型检测语义风险
        ml_result = await self.ml_filter.predict(text)
        if ml_result.risk_score > 0.8:
            return FilterResult(blocked=True, reason="high_risk")

        # 3. LLM复杂语境判断
        llm_result = await self.llm_judge.judge(text)
        return FilterResult(
            blocked=llm_result.should_block,
            risk_level=llm_result.risk_level
        )
```

#### 最新关系抽取方案

```python
# LLM辅助关系抽取架构
class RelationExtractor:
    def __init__(self):
        self.llm = ClaudeClient()
        self.schema = RelationSchema()  # 预定义关系类型

    async def extract(self, text: str) -> List[Relation]:
        # 1. 实体识别
        entities = await self.llm.extract_entities(text)

        # 2. 关系候选生成
        relations = await self.llm.extract_relations(text, entities)

        # 3. 关系类型分类
        typed_relations = []
        for rel in relations:
            rel_type = await self.classify_relation_type(rel)
            if self.schema.is_valid(rel_type):
                typed_relations.append(rel.with_type(rel_type))

        # 4. 三元组去噪
        return self.deduplicate(typed_relations)
```

**适用性评估：9/10**

```
└─────────────────────────────────────────────────────────────┘
                              │
                              ▼
┌─────────────────────────────────────────────────────────────┐
│                     LLM Generation                          │
│                    (基于融合上下文生成答案)                     │

```

#### Neo4j GraphRAG库

```python
# Neo4j GraphRAG Python库使用示例
from neo4j import GraphDatabase
from neo4j_graphrag.retrievers import VectorRetriever, KGRetriever
from neo4j_graphrag.generation import GraphRAG

# 初始化
driver = GraphDatabase.driver("bolt://localhost:7687", auth=(...))

# 定义检索器
vector_retriever = VectorRetriever(driver, index_name="text_embeddings")
kg_retriever = KGRetriever(driver, index_name="entities")

# 构建GraphRAG管道
rag = GraphRAG(
    retriever=vector_retriever,
    kg_retriever=kg_retriever,
    llm=ClaudeClient()
)

# 查询
result = await rag.search("马克思主义的三个组成部分是什么？")
```

```python
class ContentVersionControl:
    def __init__(self):
        self.git = GitClient()
        self.db = VersionDB()
        self.lfs = LFSClient()

    async def save_content(self, content: Content, author: str) -> Version:
        # 1. 文本内容 → Git
        git_commit = await self.git.commit(
            path=content.text_path,
            content=content.markdown,
            author=author,
            message=content.change_summary
        )

        # 2. 二进制内容 → Git LFS
        lfs_refs = []
        for media in content.media:
            lfs_ref = await self.lfs.upload(media)
            lfs_refs.append(lfs_ref)

        # 3. 版本记录 → 数据库
        version = await self.db.create_version(
            git_commit=git_commit.sha,
            lfs_refs=lfs_refs,
            metadata=content.metadata
        )

        return version
```

```python
    def add_watermark(self, text: str) -> str:
        """生成时嵌入水印"""
        tokens = self.tokenize(text)
        watermarked = []
        for i, token in enumerate(tokens):
            if i % 3 == 0 and token in self.green_token_ids:
                # 优先选择绿名单token
                watermarked.append(self.prefer_green(token))
            else:
                watermarked.append(token)

    def detect_watermark(self, text: str) -> WatermarkResult:
        """检测水印"""
        tokens = self.tokenize(text)
        green_count = sum(1 for t in tokens if t in self.green_token_ids)
        score = green_count / len(tokens)
        return WatermarkResult(
            has_watermark=score > 0.3,
            confidence=score,
            is_ai_generated=score > 0.5
        )
```
